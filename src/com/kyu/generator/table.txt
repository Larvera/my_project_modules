SELECT A.TABLE_SCHEMA
	     , A.TABLE_NAME
	     , A.TABLE_COMMENT
	     , B.COLUMN_NAME
	     , B.DATA_TYPE
	     , B.CHARACTER_MAXIMUM_LENGTH
	     , B.COLUMN_COMMENT
	  FROM (
		SELECT TABLE_SCHEMA
		     , TABLE_NAME
		     , TABLE_COMMENT
		  FROM INFORMATION_SCHEMA.TABLES
		 WHERE TABLE_SCHEMA IN ('TAD', 'TAD_LOG')
		   AND TABLE_NAME IN ('TBL_ACCOUNT')
		) A INNER JOIN
		(
		SELECT TABLE_SCHEMA
		     , TABLE_NAME
		     , COLUMN_NAME
		     , DATA_TYPE
		     , CHARACTER_MAXIMUM_LENGTH
		     , COLUMN_COMMENT
		  FROM COLUMNS
		 WHERE TABLE_SCHEMA IN ('TAD', 'TAD_LOG')
		   AND TABLE_NAME IN ('TBL_ACCOUNT')
		) B ON A.TABLE_NAME = B.TABLE_NAME
